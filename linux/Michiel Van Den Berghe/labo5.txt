2.	#!/bin/bash
	#
	#Oef1.2
	#
	
	echo "Hoeveel scripts wil u maken?"
	read x
	for ((i=1 ; i<=$x ; i++))
	do
		echo "Geef de naam van script $i"
		read naam
		touch $naam.sh
		chmod u+x $naam.sh
	done

3.  #!/bin/bash
	#
	#Oef1.3
	#
	echo "Welk bestand wilt u wissen ?"
	read bestandsnaam
	echo "Wilt u het bestand $bestandsnaam wissen ? (Ja/Nee)"
	read antwoord
	if [ $antwoord = "Ja" ]
	then
		rm $bestandsnaam
		echo "$bestandsnaam is gewist"
	elif [ $antwoord = "Nee" ]
	then
		echo "$bestandsnaam niet verwijderd"
	else
		echo "Ongeldig antwoord (Ja/Nee)"
	fi
